seteing app via debug:

#MATIKAN LAUNCHER BAWAAN
adb -s 192.168.1.7:5555 shell pm disable-user --user 0 com.google.android.tvlauncher

#Perintah Aktifkan Kembali (Enable)
adb -s 192.168.1.7:5555 shell pm enable com.google.android.tvlauncher

#CEK APAKAH MINFEST SUDAH TERDETEKSI 
adb -s 192.168.1.7:5555 shell query-intent-activities -a android.intent.action.MAIN -c android.intent.category.HOME

#SETELAH INSTALL
adb -s 192.168.1.7:5555 shell am start -a android.intent.action.MAIN -c android.intent.category.HOME

#CEK LAUNCHER YANG SEDANG ACTIVE
adb -s 192.168.1.7:5555 shell dumpsys window | grep -E 'mCurrentFocus|mFocusedApp'
